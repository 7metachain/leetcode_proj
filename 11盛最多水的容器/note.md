```cpp
class Solution {
public:
    int maxArea(vector<int>& height) {
        int l = 0, r = height.size() - 1;
        long long ans = 0;

        while (l < r) {
            long long h = min(height[l], height[r]);
            long long w = r - l;
            ans = max(ans, h * w);

            if (height[l] < height[r])
                l++;
            else
                r--;
        }
        return (int)ans;
    }
};
```
